199 static rgb_pixel HSBtoRGB(float hue, float saturation, float brightness) {
200 
201  double hh, p, q, t, ff;
202  long i;
203  double r_out;
204  double g_out;
205  double b_out;
206 
207  if(saturation <= 0.0) { // < is bogus, just shuts up warnings
208  r_out = brightness;
209  g_out = brightness;
210  b_out = brightness;
211  return rgb_pixel(r_out * 255, g_out * 255, b_out * 255);
212  }
213  hh = hue;
214  if(hh >= 360.0) hh = 0.0;
215  hh /= 60.0;
216  i = (long)hh;
217  ff = hh - i;
218  p = brightness * (1.0 - saturation);
219  q = brightness * (1.0 - (saturation * ff));
220  t = brightness * (1.0 - (saturation * (1.0 - ff)));
221 
222  switch(i) {
223  case 0:
224  r_out = brightness;
225  g_out = t;
226  b_out = p;
227  break;
228  case 1:
229  r_out = q;
230  g_out = brightness;
231  b_out = p;
232  break;
233  case 2:
234  r_out = p;
235  g_out = brightness;
236  b_out = t;
237  break;
238 
239  case 3:
240  r_out = p;
241  g_out = q;
242  b_out = brightness;
243  break;
244  case 4:
245  r_out = t;
246  g_out = p;
247  b_out = brightness;
248  break;
249  case 5:

299 static rgbw_pixel RGBtoRGBW2(uint8_t red, uint8_t green, uint8_t blue) {
300  //Get the maximum between R, G, and B
301  float tM = std::max(red, std::max(green, blue));
302 
303  //If the maximum value is 0, immediately return pure black.
304  if(tM == 0)
305  { return rgbw_pixel(0, 0, 0, 0); }
306 
307  //This section serves to figure out what the color with 100% hue is
308  float multiplier = 255.0f / tM;
309  float hR = red * multiplier;
310  float hG = green * multiplier;
311  float hB = blue * multiplier;
312 
313  //This calculates the Whiteness (not strictly speaking Luminance) of the color
314  float M = std::max(hR, std::max(hG, hB));
315  float m = std::min(hR, std::min(hG, hB));
316  float Luminance = ((M + m) / 2.0f - 127.5f) * (255.0f/127.5f) / multiplier;
317 
318  //Calculate the output values
319  uint8_t Wo = Luminance;
320  uint8_t Bo = blue - Luminance;
321  uint8_t Ro = red - Luminance;
322  uint8_t Go = green - Luminance;
323 
324  return rgbw_pixel(Ro, Go, Bo, Wo);
325 }
